-- Make GUI frame draggable (supports mouse & touch)
local function makeDraggable(frame)
    local dragging = false
    local dragInput = nil
    local dragStart = nil
    local startPos = nil

    local function update(input)
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(
            startPos.X.Scale, startPos.X.Offset + delta.X,
            startPos.Y.Scale, startPos.Y.Offset + delta.Y
        )
    end

    frame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position

            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    frame.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            update(input)
        end
    end)
end

    label.Position = UDim2.new(1, -310, 0, 10)
    label.AnchorPoint = Vector2.new(1, 0)

    end)
end

-- Logic variables
local noStun = false

local speedConns = {}

-- Functions to apply/reset speed for No Stun feature
local function applySpeed()
    local hum = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
    if hum and hum.WalkSpeed ~= desiredSpeed then
        hum.WalkSpeed = desiredSpeed
    end
end

local function resetSpeed()
    local hum = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
    if hum then
        hum.WalkSpeed = defaultSpeed
    end
end

local function clearSpeedConns()
    for _, conn in pairs(speedConns) do
        conn:Disconnect()
    end
    speedConns = {}
end

        for _ = 1, 3 do

        end

    end
end)

-- Animation trigger without GUI for Dunked Garou effect
local function onAnimation(id, func)
    local id = tostring(id):gsub("rbxassetid://", "")

    local function connectHumanoid(humanoid)
        if humanoid then
            humanoid.AnimationPlayed:Connect(function(animationTrack)
                local vID = animationTrack.Animation.AnimationId:gsub("rbxassetid://", "")
                if id == vID then
                    func(animationTrack)
                end
            end)
        end
    end

    local char = player.Character
    connectHumanoid(char and char:FindFirstChildOfClass("Humanoid"))

    player.CharacterAdded:Connect(function(char)
        connectHumanoid(char:WaitForChild("Humanoid"))
    end)
end

local function createPopup(title, options)
    local popup = Instance.new("Frame", subscribeGui)
    popup.Size = UDim2.new(0, 350, 0, 160)
    popup.Position = UDim2.new(0.5, -175, -1, 0)
    popup.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    popup.AnchorPoint = Vector2.new(0.5, 0)
    popup.Name = "Popup"

    local titleLabel = Instance.new("TextLabel", popup)
    titleLabel.Size = UDim2.new(1, -20, 0.5, 0)
    titleLabel.Position = UDim2.new(0, 10, 0, 10)
    titleLabel.Text = title
    titleLabel.Font = Enum.Font.SourceSansBold
    titleLabel.TextSize = 24
    titleLabel.TextColor3 = Color3.new(1, 1, 1)
    titleLabel.BackgroundTransparency = 1

    local buttonContainer = Instance.new("Frame", popup)
    buttonContainer.Size = UDim2.new(1, 0, 0.4, 0)
    buttonContainer.Position = UDim2.new(0, 0, 0.6, 0)
    buttonContainer.BackgroundTransparency = 1

    for i, option in ipairs(options) do
        local btn = Instance.new("TextButton", buttonContainer)
        btn.Size = UDim2.new(0.4, 0, 0.9, 0)
        btn.Position = UDim2.new(0.05 + (i-1)*0.5, 0, 0.05, 0)
        btn.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
        btn.Text = option.text
        btn.TextColor3 = Color3.new(1, 1, 1)
        btn.Font = Enum.Font.SourceSansBold
        btn.TextSize = 20

        btn.MouseButton1Click:Connect(function()
            if option.callback then option.callback() end
            TweenService:Create(popup, TweenInfo.new(0.3, Enum.EasingStyle.Sine), {Position = UDim2.new(0.5, -175, -1, 0)}):Play()

            task.delay(0.3, function()
                popup:Destroy()
                table.remove(popupQueue, 1)
                if #popupQueue > 0 then
                    createPopup(unpack(popupQueue[1]))
                else
                    TweenService:Create(blackFade, TweenInfo.new(0.4, Enum.EasingStyle.Sine), {BackgroundTransparency = 1}):Play()
                    task.delay(0.4, function()
                        subscribeGui:Destroy()
                        if selectedDevice == "Mobile" then
                            uiToggle.Visible = true
                            notify("Use toggle button to open GUI")
                        else
                            notify("Press B to toggle GUI")
                            if inputConnection then inputConnection:Disconnect() end
                            inputConnection = UserInputService.InputBegan:Connect(function(input)
                                if input.KeyCode == Enum.KeyCode.B then
                                    uiVisible = not uiVisible
                                    toggleGUI(uiVisible)
                                end
                            end)
                        end
                    end)
                end
            end)
        end)
    end

    TweenService:Create(popup, TweenInfo.new(0.4, Enum.EasingStyle.Back), {Position = UDim2.new(0.5, -175, 0.35, 0)}):Play()

    return popup
end
